#include <stdio.h>
#include <stdlib.h>

void bubble_sort(int arr[], int n_2551){
    int i_2551, j_2551, swap_c = 0;
    for(i_2551 = 0; i_2551 < n_2551 - 1; i_2551++){
        for(j_2551 = 0; j_2551 < n_2551 - i_2551 - 1; j_2551++){
            if(arr[j_2551] > arr[j_2551 + 1]){
                int temp_2551 = arr[j_2551];
                arr[j_2551] = arr[j_2551 + 1];
                arr[j_2551 + 1] = temp_2551;
                swap_c++;
            }
        }
    }
    printf("No. of swaps: %d\n", swap_c);
    printf("Sorted array: \n");
    for(int i_2551 = 0; i_2551 < n_2551; i_2551++){
        printf("%d ", arr[i_2551]);
    }
    printf("\n");
}

void insertion_sort(int arr[], int n){
    int i_2551, j_2551, swap_c = 0;
    for(i_2551 = 1; i_2551 < n; i_2551++){
        int key_2551 = arr[i_2551];
        j_2551 = i_2551 - 1;
        while(j_2551 >= 0 && arr[j_2551] > key_2551){
            arr[j_2551 + 1] = arr[j_2551];
            j_2551 = j_2551 - 1;
            swap_c++;
        }
        arr[j_2551 + 1] = key_2551;
    }
    printf("No. of swaps: %d\n", swap_c);
    printf("Sorted array: \n");
    for(int i_2551 = 0; i_2551 < n; i_2551++){
        printf("%d ", arr[i_2551]);
    }
    printf("\n");
}

// helper recursive selection sort and a global swap counter
int sel_swap_count_2551 = 0;

void selection_sort_rec(int arr[], int n, int index){
    if(index == n-1){
        return;
    }
    int min_index_2551 = index;

    for(int i = index + 1; i < n; i++){
        if(arr[i] < arr[min_index_2551]){
            min_index_2551 = i;
        }
    }

    if(min_index_2551 != index){
        int temp = arr[index];
        arr[index] = arr[min_index_2551];
        arr[min_index_2551] = temp;
        sel_swap_count_2551++;
    }

    selection_sort_rec(arr, n, index + 1);
}

// wrapper selection_sort (keeps original signature)
void selection_sort(int arr[], int n, int index){
    sel_swap_count_2551 = 0;
    selection_sort_rec(arr, n, index);

    printf("No. of swaps: %d\n", sel_swap_count_2551);
    printf("Sorted array: \n");
    for(int i_2551 = 0; i_2551 < n; i_2551++){
        printf("%d ", arr[i_2551]);
    }
    printf("\n");
}

// Recursion method for quick sort (no printing inside)
void quick_sort(int arr[], int low, int high){
    if(low < high){
        int pivot = arr[high];
        int i_2551 = (low - 1);
        for(int j_2551 = low; j_2551 < high; j_2551++){
            if(arr[j_2551] < pivot){
                i_2551++;
                int temp_2551 = arr[i_2551];
                arr[i_2551] = arr[j_2551];
                arr[j_2551] = temp_2551;
            }
        }
        int temp_2551 = arr[i_2551 + 1];
        arr[i_2551 + 1] = arr[high];
        arr[high] = temp_2551;
        int pi_2551 = i_2551 + 1;

        quick_sort(arr, low, pi_2551 - 1);
        quick_sort(arr, pi_2551 + 1, high);
    }
}
